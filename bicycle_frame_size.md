# Bicycle frame size logic

```math
# Helper function to get the number of rows in the data matrix
nrows(m) = subset(size(m), index(1))

# Helper to extract a specific column from the data matrix
col(m, c) = subset(m, index(range(1, nrows(m)), c))

# Helper to calculate weights: closer points = higher weights (1 / distance^2)
get_weights(data, h, i) = 1 ./ ((col(data, 1) - h).^2 + (col(data, 2) - i).^2 + 1e-6)

# Main function to interpolate frame size
# Returns a decimal frame size (e.g., 53.4) based on the weighted average of all data points
interpolate_frame(data, h, i) = sum(get_weights(data, h, i) .* col(data, 3)) / sum(get_weights(data, h, i))

# Function to find triplet(s) for a specific height and inseam
# data: The matrix of triplets
# h: Height (metric)
# i: Inseam (metric)
find_triplets(data, h, i) = subset(data, index(filter(range(1, nrows(data)), r = subset(data, index(r, 1)) == h and subset(data, index(r, 2)) == i), range(1, 3)))

# Function to find triplet(s) within a range of heights and inseams
# data: The matrix of triplets
# h_min, h_max: Range of heights (metric)
# i_min, i_max: Range of inseams (metric)
find_triplets_range(data, h_min, h_max, i_min, i_max) = subset(data, index(filter(range(1, nrows(data)), r = subset(data, index(r, 1)) >= h_min and subset(data, index(r, 1)) <= h_max and subset(data, index(r, 2)) >= i_min and subset(data, index(r, 2)) <= i_max), range(1, 3)))
```
